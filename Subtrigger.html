def run(repository_path, revision_number)
  Template.parse(DATA.read)
  rev = Revision.new(
    revision_number,
    svnlook('info', repository_path),
    svnlook('dirs-changed', repository_path)
  )
  Rule.matching(rev).each { |r| r.run(rev) }
end
